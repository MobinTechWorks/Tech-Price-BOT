# این تابع هنگام ارسال /start اجرا میشه
async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):

        "سلام! 👋\nبه ربات استعلام قیمت طلا و ارز خوش اومدی.\nدستور زیر رو می‌تونی بزنی:\n\n"

# این تابع بعداً قراره قیمت‌ها رو بیاره
